// lib/gamification/xp-calculator.ts
export class XPCalculator {
  // Base XP values
  private static readonly XP_VALUES = {
    SALE_COMPLETED: 50,
    REVIEW_5_STAR: 25,
    RESPONSE_UNDER_1H: 10,
    // ... etc
  };

  // Calculate XP for action
  static calculateXP(
    action: string,
    user: User,
    metadata?: any
  ): number {
    const baseXP = this.XP_VALUES[action] || 0;
    const multiplier = this.getMultiplier(user);

    return Math.floor(baseXP * multiplier);
  }

  // Get user's current multiplier
  private static getMultiplier(user: User): number {
    let multiplier = 1.0;

    // Streak multiplier
    if (user.streakDays >= 90) multiplier *= 3.0;
    else if (user.streakDays >= 30) multiplier *= 2.0;
    else if (user.streakDays >= 14) multiplier *= 1.5;
    else if (user.streakDays >= 7) multiplier *= 1.2;

    // Tier multiplier
    switch (user.tier) {
      case 'PLATINUM': multiplier *= 1.5; break;
      case 'GOLD': multiplier *= 1.25; break;
      case 'SILVER': multiplier *= 1.1; break;
    }

    return multiplier;
  }

  // Calculate level from XP
  static calculateLevel(xp: number): number {
    const BASE_XP = 100;
    const EXPONENT = 1.5;

    // Reverse of: xp = BASE_XP * (level ^ EXPONENT)
    // level = (xp / BASE_XP) ^ (1 / EXPONENT)
    return Math.floor(Math.pow(xp / BASE_XP, 1 / EXPONENT)) + 1;
  }

  // Calculate XP needed for next level
  static xpForNextLevel(currentLevel: number): number {
    const BASE_XP = 100;
    const EXPONENT = 1.5;

    return Math.floor(BASE_XP * Math.pow(currentLevel, EXPONENT));
  }
}
