  action: string,
  metadata: Record<string, any>
): number {
  let xp = XP_CONFIG.baseXP;

  // Apply base multiplier
  if (XP_CONFIG.multipliers[action]) {
    xp *= XP_CONFIG.multipliers[action];
  }

  // Streak bonus

    xp *= (1 + XP_CONFIG.multipliers.streakBonus * metadata.streakDays);

  // First time bonus

  }

  return Math.round(xp);
}

// lib/gamification/level-calculator.ts

// Exponential XP curve: cada level requer 20% mais XP
export function getXPForLevel(level: number): number {
  const baseXP = 100;
  return Math.round(baseXP * Math.pow(1.2, level - 1));
}

// Total XP needed para atingir level
export function getTotalXPForLevel(level: number): number {
  let total = 0;
  for (let i = 1; i < level; i++) {
    total += getXPForLevel(i);
  }
  return total;
}

// Determinar level baseado em total XP
export function getLevelFromXP(totalXP: number): number {
  let level = 1;
  let xpRequired = 0;

  while (xpRequired <= totalXP) {
    level++;
    xpRequired += getXPForLevel(level);
  }

  return level - 1;
}

// Progress para pr√≥ximo level (0-100%)
export function getProgressToNextLevel(totalXP: number): number {
  const currentLevel = getLevelFromXP(totalXP);
  const xpForCurrentLevel = getTotalXPForLevel(currentLevel);
  const xpForNextLevel = getTotalXPForLevel(currentLevel + 1);

  const currentLevelXP = totalXP - xpForCurrentLevel;
  const xpNeeded = xpForNextLevel - xpForCurrentLevel;

  return Math.round((currentLevelXP / xpNeeded) * 100);
}
